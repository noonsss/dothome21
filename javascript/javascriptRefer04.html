01. if문 

{ 
    //true, 1, "1", [ ], { } > true
    //false, 0, null, undefined,"" > false

    if( true ){
        document.write("조건문이 실행되었습니다.(true)");
    } else {
        document.write("조건문이 실행되었습니다.(false)");
    }
}

02. 다중if문 

{ 
    let num = 100;

    if ( num == 90 ){
        document.write("조건문이 실행되었습니다.(num=90)");
    } else if ( num == 100 ){
        document.write("조건문이 실행되었습니다.(num=100)");
    } else if ( num == 200 ){
        document.write("조건문이 실행되었습니다.(num=200)");
    } else if ( num == 300 ){
        document.write("조건문이 실행되었습니다.(num=300)");
    } else {
        document.write("조건문이 실행되었습니다.(num == 값이 없음)");
    }
}

03. 중첩if문 

{ 
    let num = 100;

    if( num == 100 ){
        document.write("조건문이 실행되었습니다.(1)");
        if( num == 100 ){
            document.write("조건문이 실행되었습니다.(2)");
            if( num == 100){
                document.write("조건문이 실행되었습니다.(3)");
            }
        }
    } else {
        document.write("조건문이 실행되었습니다.(4)");
    }
}

04. switch문 

{ 
    let num = 100;

    switch( num ){
        case 100 : 
            document.write("조건문이 실행되었습니다.(num = 100)

");
        break;
        case 200 : 
            document.write("조건문이 실행되었습니다.(num = 200)
");
        break;
        case 300 : 
            document.write("조건문이 실행되었습니다.(num = 300)
");
        break;
        default :  
            document.write("조건문이 실행되었습니다.(num = 값이 없음)
");
        break;
    }
}

07. while문 

{ 
    let num = 1;
    while( num <= 10 ){
        document.write( num + ". 반복문이 실행되었습니다.
");
        num++;  //num = num + 1; 과 같다, =이 하나 있으면 오른쪽이 왼쪽에 대입
    }
}

08. do while문 

{ 
    let num = 1;
    do {
        document.write( num + ".반복문이 실행되었습니다." );
        num++;
    } while ( num <= 10 );
}

09. for문 

{ 
    //for(초깃값; 조건식; 증감값){
    //    //실행문

    for( let i=1; i<=10; i++ ){

    }
}

10. 중첩 for문 

{ 
    for ( var i=1; i<=5; i++ ){
        document.write( i + ". 반복문(아이)이 실행되었습니다." );
        for ( var j=1; j<=5; j++ ){
            document.write( j + ". 반복문(제이)이 실행되었습니다." );
        }
    }
}

11. forEach( ) 

{ 
    const num = [100, 200, 300, 400, 500];

    document.write(num[0]);
    document.write(num[1]);
    document.write(num[2]);
    document.write(num[3]);
    document.write(num[4]);

    //for문
    for ( let i=0; i<num.length; i++ ){
        document.write(num[i]);
    }

    //forEach( )
    num.forEach(function(el){
        document.write(el);
    });

    //forEach(element, index, array) - 숫자, 순서, 배열
    num.forEach((element, index, array) => {
        document.write(element);
        document.write(index);
        document.write(array);
    });               
}

12. for of문 

{ 
    const num = [100, 200, 300, 400, 500];

    for( let i of num ) {
        document.write( i + ". 반복문이 실행되었습니다." );
    }
}

13. for in문 

{ 
    //배열
    const num = [100, 200, 300, 400, 500];

    for( let i in num ) {
        document.write(num[i]);
    }

    //객체
    const obj = {
        a: 100,
        b: 200,
        c: 300,
        d: 400,
        e: 500
    }

    for( let key in obj ){
        document.write(key);
        document.write(obj[key]);
    }
}

14. break문 

{ 
    //while문을 이용해 1부터 20까지 출력
    
    let num = 0;
    while( num < 20 ){
        num++;
        document.write(num);
        if( num == 10 ){
            break;
        }
    }
    
    //for문을 이용해 1부터 20까지 출력
    
    for ( i=1; i<=20; i++ ){
        document.write(i)
        if( i == 10 ){
            break;
        }
    }
}

15. continue문 

{ 
    let num = 0;
    while( true ){
        num++;
        if( num == 3 ){
            continue;
        }
        if( num > 19 ){
            break;
        }
        document.write(num)
    }            
}
